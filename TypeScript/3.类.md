## ts的类的定义
```ts
    class Hyc{
        name:string;//这里不加会报错
        age:number;
        constructor(name:string,age:number){
            this.name = name;
            this.age = age;
        }
        getAge():number{
            return this.age;
        }
    }
    let h = new Hyc('huang',20);
```
